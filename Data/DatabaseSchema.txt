CREATE DATABASE "HappinessProject"
    WITH 
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'English_United States.1252'
    LC_CTYPE = 'English_United States.1252'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1;
	
CREATE TABLE WorldHappinessReport_History
(
	country_name varchar(255),
	year_num Int,
	life_ladder float,
	log_GDP_Per_Cap float,
	Social_Support float,
	Healty_Life_Exp float,
	Freedom float,
	Generosity float,
	Percept_Corrupt float,
	Positive_Affect float,
	Negative_Affect float
);

CREATE TABLE WorldHappinessReport_2021
(
	Country_Name varchar(255),
	Regional_Indicator varchar(255),
	Ladder_Score float,
	Stand_Err float,
	Up_Whisker float,
	Low_Whisker float,
	Log_GDP_Per_Cap float,
	Social_Support float,
	Healty_Life_Exp float,
	Freedom float,
	Generosity float,
	Percept_Corrupt float,	
	Ladder_Score_Distopia float,
	ExpBy_Log_GDP_Per_Cap float,
	ExpBy_Social_Support float,
	ExpBy_Healty_Life_Exp float,
	ExpBy_Freedom float,
	ExpBy_Generosity float,
	ExpBy_Percept_Corrupt float,	
	Distopia_Plus_Resid float
);
